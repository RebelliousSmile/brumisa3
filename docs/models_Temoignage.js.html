const BaseModel = require('./BaseModel');

/**
 * Modèle Témoignage
 */
class Temoignage extends BaseModel {
    constructor() {
        super('temoignages', 'id');
        
        // Champs autorisés pour mass assignment
        this.fillable = [
            'auteur_nom', 'auteur_email', 'contenu', 'note', 'systeme_jeu',
            'statut', 'ip_adresse', 'user_agent', 'source', 'verifie'
        ];
        
        // Champs protégés
        this.guarded = ['id', 'date_creation', 'date_modification'];
        
        // Champs cachés lors de la sérialisation publique
        this.hidden = ['auteur_email', 'ip_adresse', 'user_agent'];
        
        // Conversions automatiques
        this.casts = {
            note: 'integer',
            verifie: 'boolean'
        };
        
        this.timestamps = true;
    }

    /**
     * Validation métier
     */
    async validate(data, operation = 'create') {
        const erreurs = [];

        // Nom auteur requis
        if (!data.auteur_nom || data.auteur_nom.trim().length < 2) {
            erreurs.push('Nom de l\'auteur requis (minimum 2 caractères)');
        }

        // Email valide si fourni
        if (data.auteur_email && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(data.auteur_email)) {
            erreurs.push('Format email invalide');
        }

        // Contenu requis
        if (!data.contenu || data.contenu.trim().length < 10) {
            erreurs.push('Contenu requis (minimum 10 caractères)');
        }

        // Contenu pas trop long
        if (data.contenu && data.contenu.length > 1000) {
            erreurs.push('Contenu trop long (maximum 1000 caractères)');
        }

        // Note valide
        if (data.note !== undefined && (data.note < 1 || data.note > 5)) {
            erreurs.push('Note doit être entre 1 et 5');
        }

        // Système de jeu valide si fourni
        const systemesValides = ['monsterhearts', 'engrenages', 'metro2033', 'mistengine', 'general'];
        if (data.systeme_jeu && !systemesValides.includes(data.systeme_jeu)) {
            erreurs.push('Système de jeu invalide');
        }

        // Statut valide
        const statutsValides = ['EN_ATTENTE', 'APPROUVE', 'REJETE', 'MASQUE'];
        if (data.statut && !statutsValides.includes(data.statut)) {
            erreurs.push('Statut invalide');
        }

        if (erreurs.length > 0) {
            throw new Error(`Erreurs de validation: ${erreurs.join(', ')}`);
        }

        return true;
    }

    /**
     * Définit les valeurs par défaut avant création
     */
    async beforeCreate(data) {
        // Statut par défaut
        if (!data.statut) {
            data.statut = 'EN_ATTENTE';
        }

        // Système général par défaut
        if (!data.systeme_jeu) {
            data.systeme_jeu = 'general';
        }

        // Vérifié par défaut à false
        if (data.verifie === undefined) {
            data.verifie = false;
        }

        // Source par défaut
        if (!data.source) {
            data.source = 'site_web';
        }

        // Nettoyer le contenu
        data.contenu = this.nettoyerContenu(data.contenu);
        data.auteur_nom = this.nettoyerNom(data.auteur_nom);

        return data;
    }

    /**
     * Nettoie le contenu du témoignage
     */
    nettoyerContenu(contenu) {
        return contenu
            .trim()
            .replace(/\s+/g, ' ')
            .substring(0, 1000);
    }

    /**
     * Nettoie le nom de l'auteur
     */
    nettoyerNom(nom) {
        return nom
            .trim()
            .replace(/[<>]/g, '')
            .substring(0, 100);
    }

    /**
     * Trouve les témoignages approuvés
     */
    async findApprouves(systeme = null, limite = 10) {
        let where = 'statut = ?';
        let params = ['APPROUVE'];

        if (systeme && systeme !== 'general') {
            where += ' AND (systeme_jeu = ? OR systeme_jeu = ?)';
            params.push(systeme, 'general');
        }

        return await this.findAll(where, params, 'date_creation DESC', limite);
    }

    /**
     * Trouve les témoignages en attente de modération
     */
    async findEnAttente(limite = 50) {
        return await this.findAll(
            'statut = ?',
            ['EN_ATTENTE'],
            'date_creation ASC',
            limite
        );
    }

    /**
     * Approuve un témoignage
     */
    async approuver(id, moderateurId = null) {
        const temoignage = await this.findById(id);
        if (!temoignage) {
            throw new Error('Témoignage non trouvé');
        }

        if (temoignage.statut !== 'EN_ATTENTE') {
            throw new Error('Seuls les témoignages en attente peuvent être approuvés');
        }

        const updateData = { 
            statut: 'APPROUVE',
            verifie: true
        };

        if (moderateurId) {
            updateData.moderateur_id = moderateurId;
        }

        return await this.update(id, updateData);
    }

    /**
     * Rejette un témoignage
     */
    async rejeter(id, raison = null, moderateurId = null) {
        const temoignage = await this.findById(id);
        if (!temoignage) {
            throw new Error('Témoignage non trouvé');
        }

        const updateData = { 
            statut: 'REJETE'
        };

        if (raison) {
            updateData.raison_rejet = raison;
        }

        if (moderateurId) {
            updateData.moderateur_id = moderateurId;
        }

        return await this.update(id, updateData);
    }

    /**
     * Masque un témoignage approuvé
     */
    async masquer(id, raison = null) {
        const temoignage = await this.findById(id);
        if (!temoignage) {
            throw new Error('Témoignage non trouvé');
        }

        const updateData = { 
            statut: 'MASQUE'
        };

        if (raison) {
            updateData.raison_masquage = raison;
        }

        return await this.update(id, updateData);
    }

    /**
     * Calcule la note moyenne
     */
    async noteMoyenne(systeme = null) {
        let where = 'statut = ? AND note IS NOT NULL';
        let params = ['APPROUVE'];

        if (systeme && systeme !== 'general') {
            where += ' AND (systeme_jeu = ? OR systeme_jeu = ?)';
            params.push(systeme, 'general');
        }

        const sql = `
            SELECT AVG(CAST(note AS FLOAT)) as moyenne, COUNT(*) as nombre
            FROM ${this.tableName} 
            WHERE ${where}
        `;
        
        const db = require('../database/db');
        const result = await db.get(sql, params);
        
        return {
            moyenne: result.moyenne ? Math.round(result.moyenne * 10) / 10 : 0,
            nombre: result.nombre || 0
        };
    }

    /**
     * Statistiques par système
     */
    async statistiquesParSysteme() {
        const sql = `
            SELECT 
                systeme_jeu,
                COUNT(*) as total,
                COUNT(CASE WHEN statut = 'APPROUVE' THEN 1 END) as approuves,
                COUNT(CASE WHEN statut = 'EN_ATTENTE' THEN 1 END) as en_attente,
                AVG(CASE WHEN note IS NOT NULL AND statut = 'APPROUVE' THEN CAST(note AS FLOAT) END) as note_moyenne
            FROM ${this.tableName} 
            GROUP BY systeme_jeu
            ORDER BY approuves DESC
        `;
        
        const db = require('../database/db');
        return await db.all(sql);
    }

    /**
     * Recherche dans les témoignages
     */
    async rechercher(query, systeme = null, statut = 'APPROUVE') {
        let where = 'statut = ? AND (contenu LIKE ? OR auteur_nom LIKE ?)';
        let params = [statut, `%${query}%`, `%${query}%`];

        if (systeme && systeme !== 'general') {
            where += ' AND (systeme_jeu = ? OR systeme_jeu = ?)';
            params.push(systeme, 'general');
        }

        return await this.findAll(where, params, 'date_creation DESC', 20);
    }

    /**
     * Obtient les témoignages récents pour la page d'accueil
     */
    async obtenirPourAccueil(nombre = 6) {
        return await this.findAll(
            'statut = ? AND note >= ?',
            ['APPROUVE', 4],
            'date_creation DESC',
            nombre
        );
    }

    /**
     * Vérifie si un utilisateur peut poster un témoignage
     */
    async peutPoster(ipAdresse, email = null) {
        // Limite par IP (1 par jour)
        const hier = new Date();
        hier.setDate(hier.getDate() - 1);
        
        const parIp = await this.count(
            'ip_adresse = ? AND date_creation > ?',
            [ipAdresse, hier.toISOString()]
        );

        if (parIp > 0) {
            return { 
                autorise: false, 
                raison: 'Un témoignage par IP par jour maximum' 
            };
        }

        // Limite par email si fourni (1 par semaine)
        if (email) {
            const semainePassee = new Date();
            semainePassee.setDate(semainePassee.getDate() - 7);
            
            const parEmail = await this.count(
                'auteur_email = ? AND date_creation > ?',
                [email, semainePassee.toISOString()]
            );

            if (parEmail > 0) {
                return { 
                    autorise: false, 
                    raison: 'Un témoignage par email par semaine maximum' 
                };
            }
        }

        return { autorise: true };
    }

    /**
     * Nettoie les anciens témoignages rejetés
     */
    async nettoyerAnciens(joursAnciennete = 30) {
        const dateLimit = new Date();
        dateLimit.setDate(dateLimit.getDate() - joursAnciennete);
        
        const result = await this.delete(
            'statut = ? AND date_creation < ?',
            ['REJETE', dateLimit.toISOString()]
        );
        
        return result;
    }
}

module.exports = Temoignage;
