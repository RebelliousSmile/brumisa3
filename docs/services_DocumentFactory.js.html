// Documents par type
const GenericDocument = require('./documents/GenericDocument');
const ClassPlanDocument = require('./documents/ClassPlanDocument');
const CharacterSheetDocument = require('./documents/CharacterSheetDocument');
const TownDocument = require('./documents/TownDocument');
const GroupDocument = require('./documents/GroupDocument');
const OrganizationDocument = require('./documents/OrganizationDocument');
const DangerDocument = require('./documents/DangerDocument');

// Thèmes par système
const MonsterheartsTheme = require('./themes/MonsterheartsTheme');
const ZombiologyTheme = require('./themes/ZombiologyTheme');
const EngrenagesTheme = require('./themes/EngrenagesTheme');
const Metro2033Theme = require('./themes/Metro2033Theme');
const MistEngineTheme = require('./themes/MistEngineTheme');

/**
 * Factory pour créer les documents PDF avec la bonne architecture
 * Document Type + System Theme
 */
class DocumentFactory {
    
    /**
     * Crée un document selon le type et le système
     * Supporte les 6 types de documents selon architecture-models.md
     * 
     * @param {string} documentType - Type de document (CHARACTER, TOWN, GROUP, ORGANIZATION, DANGER, GENERIQUE)
     * @param {string} system - Système de jeu ('monsterhearts', 'engrenages', 'metro2033', 'mistengine', 'zombiology')
     * @param {Object} options - Options supplémentaires
     * @returns {Object} Instance du document
     */
    static createDocument(documentType, system, options = {}) {
        // Normaliser le type de document
        const normalizedType = documentType.toUpperCase();
        
        // Obtenir le thème du système
        const theme = this.getSystemTheme(system);
        
        // Créer le document selon le type
        switch (normalizedType) {
            case 'CHARACTER':
                if (!theme) {
                    throw new Error('CharacterDocument nécessite un thème système');
                }
                return new CharacterSheetDocument(theme, options);
                
            case 'TOWN':
                if (!theme) {
                    throw new Error('TownDocument nécessite un thème système');
                }
                return new TownDocument(theme, options);
                
            case 'GROUP':
                if (!theme) {
                    throw new Error('GroupDocument nécessite un thème système');
                }
                return new GroupDocument(theme, options);
                
            case 'ORGANIZATION':
                if (!theme) {
                    throw new Error('OrganizationDocument nécessite un thème système');
                }
                return new OrganizationDocument(theme, options);
                
            case 'DANGER':
                if (!theme) {
                    throw new Error('DangerDocument nécessite un thème système');
                }
                return new DangerDocument(theme, options);
                
            case 'GENERIQUE':
            case 'GENERIC':
                return new GenericDocument(theme, options);
                
            // Support des anciens types pour compatibilité
            case 'CLASS-PLAN':
                if (!theme) {
                    throw new Error('ClassPlanDocument nécessite un thème système');
                }
                return new ClassPlanDocument(theme, options);
                
            case 'CHARACTER-SHEET':
                if (!theme) {
                    throw new Error('CharacterSheetDocument nécessite un thème système');
                }
                return new CharacterSheetDocument(theme, options);
                
            default:
                throw new Error(`Type de document non supporté: ${documentType}. Types supportés: CHARACTER, TOWN, GROUP, ORGANIZATION, DANGER, GENERIQUE`);
        }
    }
    
    /**
     * Crée un document à partir d'un template legacy
     * @param {string} system - Système de jeu
     * @param {string} template - Template legacy
     * @returns {Object} Instance du document
     */
    static createFromTemplate(system, template) {
        // Mapper les anciens templates vers la nouvelle architecture
        const mapping = this.getTemplateMapping(system, template);
        
        return this.createDocument(mapping.documentType, system);
    }
    
    /**
     * Obtient le thème pour un système donné
     * Supporte les 5 systèmes JDR selon architecture-models.md
     * 
     * @param {string} system - Système de jeu
     * @returns {SystemTheme|null} Instance du thème ou null
     */
    static getSystemTheme(system) {
        // Normaliser le nom du système
        const normalizedSystem = system?.toLowerCase();
        
        try {
            switch (normalizedSystem) {
                case 'monsterhearts':
                    return new MonsterheartsTheme();
                    
                case 'engrenages':
                    return new EngrenagesTheme();
                    
                case 'metro2033':
                case 'metro':
                    return new Metro2033Theme();
                    
                case 'mistengine':
                case 'mist':
                    return new MistEngineTheme();
                    
                case 'zombiology':
                    return new ZombiologyTheme();
                    
                case null:
                case undefined:
                case '':
                    // Document générique sans thème spécifique
                    return null;
                    
                default:
                    throw new Error(`Système non supporté: ${system}. Systèmes supportés: monsterhearts, engrenages, metro2033, mistengine, zombiology`);
            }
        } catch (requireError) {
            // Gérer le cas où un thème n'est pas encore implémenté
            if (requireError.code === 'MODULE_NOT_FOUND') {
                console.warn(`⚠️ Thème ${system} pas encore implémenté, utilisation du thème générique`);
                return null;
            }
            throw requireError;
        }
    }

    /**
     * Vérifier si un type de document est supporté par un système
     * 
     * @param {string} documentType - Type de document
     * @param {string} system - Système de jeu
     * @returns {boolean} Supporté ou non
     */
    static isDocumentTypeSupported(documentType, system) {
        const normalizedType = documentType.toUpperCase();
        const normalizedSystem = system?.toLowerCase();
        
        // Configuration des types supportés par système
        const supportedTypes = {
            'monsterhearts': ['CHARACTER', 'TOWN', 'GROUP', 'GENERIQUE'],
            'engrenages': ['CHARACTER', 'ORGANIZATION', 'GENERIQUE'],
            'metro2033': ['CHARACTER', 'ORGANIZATION', 'DANGER', 'GENERIQUE'],
            'mistengine': ['CHARACTER', 'DANGER', 'GENERIQUE'],
            'zombiology': ['CHARACTER', 'ORGANIZATION', 'DANGER', 'GENERIQUE']
        };
        
        return supportedTypes[normalizedSystem]?.includes(normalizedType) || false;
    }

    /**
     * Mappe les anciens templates vers les nouveaux types de documents
     * @param {string} system - Système de jeu
     * @param {string} template - Template legacy
     * @returns {Object} Mapping vers le nouveau type
     */
    static getTemplateMapping(system, template) {
        // Mappings spécifiques par système
        const mappings = {
            monsterhearts: {
                'plan-classe-instructions': { documentType: 'class-plan' },
                'plan-classe-instructions-test': { documentType: 'class-plan' },
                'document-generique': { documentType: 'generic' },
                'document-generique-v2': { documentType: 'generic' },
                'feuille-personnage': { documentType: 'character-sheet' }
            },
            engrenages: {
                'document-generique': { documentType: 'generic' },
                'feuille-personnage': { documentType: 'character-sheet' }
            },
            metro2033: {
                'document-generique': { documentType: 'generic' },
                'feuille-personnage': { documentType: 'character-sheet' }
            },
            mistengine: {
                'document-generique': { documentType: 'generic' },
                'feuille-personnage': { documentType: 'character-sheet' }
            },
            zombiology: {
                'document-generique': { documentType: 'generic' },
                'feuille-personnage': { documentType: 'character-sheet' }
            }
        };
        
        const systemMappings = mappings[system];
        if (!systemMappings) {
            throw new Error(`Aucun mapping trouvé pour le système: ${system}`);
        }
        
        const mapping = systemMappings[template];
        if (!mapping) {
            throw new Error(`Template non supporté pour ${system}: ${template}`);
        }
        
        return mapping;
    }
    
    /**
     * Liste tous les types de documents disponibles
     * @returns {Array} Liste des types
     */
    static getAvailableDocumentTypes() {
        return ['generic', 'class-plan', 'character-sheet'];
    }
    
    /**
     * Liste tous les systèmes supportés
     * @returns {Array} Liste des systèmes
     */
    static getSupportedSystems() {
        return ['monsterhearts', 'engrenages', 'metro2033', 'mistengine', 'zombiology'];
    }
    
    /**
     * Vérifie si un système est supporté
     * @param {string} system - Système à vérifier
     * @returns {boolean} true si supporté
     */
    static isSystemSupported(system) {
        return this.getSupportedSystems().includes(system);
    }
    
    /**
     * Vérifie si un type de document est supporté
     * @param {string} documentType - Type à vérifier
     * @returns {boolean} true si supporté
     */
    static isDocumentTypeSupported(documentType) {
        return this.getAvailableDocumentTypes().includes(documentType);
    }

}
